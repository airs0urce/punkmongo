var deleteAnim = gsap.timeline({});
deleteAnim.to(
    document.querySelectorAll('.deleted'), 
    {xPercent: 120, duration: 0.3}
);
deleteAnim.to(
    document.querySelectorAll('.deleted'), 
    {height: '0', marginBottom: '0', border: '0px',duration: 0.3},
    '>'
);
